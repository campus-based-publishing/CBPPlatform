00398    /**
00399     * Take an NLM preliminary meta-data array and fix publisher-loc
00400     * and publisher-name entries:
00401     * - If there is a location but no name then try to extract a
00402     *   publisher name from the location string.
00403     * - Make sure that location and name are not the same.
00404     * - Copy institution to publisher if no publisher is set,
00405     *   otherwise leave the institution.
00406     * @param $metadata array
00407     * @return array
00408     */
00409    function &fixPublisherNameAndLocation(&$metadata) {
00410       if (isset($metadata['publisher-loc'])) {
00411          // Extract publisher-name from publisher-loc if we don't have a
00412          // publisher-name in the parsing result.
00413          if (empty($metadata['publisher-name'])) {
00414             $metadata['publisher-name'] = String::regexp_replace('/.*:([^,]+),?.*/', '\1', $metadata['publisher-loc']);
00415          }
00416 
00417          // Remove publisher-name from publisher-loc
00418          $metadata['publisher-loc'] = String::regexp_replace('/^(.+):.*/', '\1', $metadata['publisher-loc']);
00419 
00420          // Check that publisher-name and location are not the same
00421          if (!empty($metadata['publisher-name']) && $metadata['publisher-name'] == $metadata['publisher-loc']) unset($metadata['publisher-name']);
00422       }
00423 
00424       // Copy the institution property (if any) as the publisher-name
00425       if (isset($metadata['institution']) &&
00426             (!isset($metadata['publisher-name']) || empty($metadata['publisher-name']))) {
00427          $metadata['publisher-name'] = $metadata['institution'];
00428       }
00429 
00430       // Clean the result
00431       foreach(array('publisher-name', 'publisher-loc') as $publisherProperty) {
00432          if (isset($metadata[$publisherProperty])) {
00433             $metadata[$publisherProperty] = String::trimPunctuation($metadata[$publisherProperty]);
