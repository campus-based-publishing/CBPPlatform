00518    /**
00519     * Take a structured meta-data statement and transform it into a
00520     * plain text value that can be displayed to the end-user.
00521     *
00522     * @param $property MetadataProperty
00523     * @param $value mixed
00524     * @return string
00525     */
00526    function _getStringValueFromMetadataStatement(&$property, &$value) {
00527       if ($property->getCardinality() == METADATA_PROPERTY_CARDINALITY_MANY && !empty($value)) {
00528          $allowedTypes = $property->getAllowedTypes();
00529          if (isset($allowedTypes[METADATA_PROPERTY_TYPE_COMPOSITE])) {
00530             // We currently only can transform composite
00531             // name arrays to strings.
00532             $allowedAssocTypes = $allowedTypes[METADATA_PROPERTY_TYPE_COMPOSITE];
00533             assert(in_array(ASSOC_TYPE_AUTHOR, $allowedAssocTypes) || in_array(ASSOC_TYPE_EDITOR, $allowedAssocTypes));
00534             import('lib.pkp.classes.metadata.nlm.NlmNameSchemaPersonStringFilter');
00535             $personStringFilter = new NlmNameSchemaPersonStringFilter(PERSON_STRING_FILTER_MULTIPLE);
00536             assert($personStringFilter->supportsAsInput($value));
00537             $stringValue = $personStringFilter->execute($value);
00538          } else {
00539             // We currently can't transform properties of
00540             // cardinality "many" to strings.
00541             assert(is_array($value) && count($value) <= 1);
00542             $stringValue = $value[0];
00543          }
00544       } else {
00545          $stringValue = (string)$value;
00546       }
00547 
00548       return $stringValue;
