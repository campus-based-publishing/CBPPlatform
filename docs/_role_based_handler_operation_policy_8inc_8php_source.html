00063    function effect() {
00064       // We implement two different role checking algorithms.
00065       if ($this->hasAuthorizedContextObject(ASSOC_TYPE_USER_GROUP)) {
00066          // 1) If there is an authorized user group in the authorization
00067          //    context then we'll make sure that that user group's
00068          //    role is one of the allowed roles.
00069          //    This has two advantages:
00070          //    - It is fast as it can be done without database access.
00071          //    - It makes sure that the selected user group belongs to one of
00072          //      the authorized roles and that we deny access if that's not
00073          //      the case.
00074          $userGroup =& $this->getAuthorizedContextObject(ASSOC_TYPE_USER_GROUP);
00075          if (!$this->_checkUserGroupRoleAssignment($userGroup)) return AUTHORIZATION_DENY;
00076       } else {
00077          // 2) If there is no authorized user group in the authorization
00078          //    context then we'll check whether the user has one of the
00079          //    allowed roles assigned. If that's the case we'll permit access.
00080          $request =& $this->getRequest();
00081          $user =& $request->getUser();

00083          if (!$this->_checkUserRoleAssignment($user)) return AUTHORIZATION_DENY;

00090          assert($this->getOperations() === array());
