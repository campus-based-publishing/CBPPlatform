00013  *
00014  *  We are using an RPC style URL-to-endpoint mapping. Our approach follows
00015  *  a simple "convention-over-configuration" paradigm. If necessary the
00016  *  router can be subclassed to implement more complex URL-to-endpoint mappings.
00017  *
00018  *  For servers with path info enabled the component URL has the following elements:
00019  *
00020  *       .../index.php/context1/context2/$$$call$$$/path/to/handler-class/operation-name?arg1=...&arg2=...
00021  *
00022  *  where "$$$call$$$" is a non-mutable literal string and "path/to" is
00023  *  by convention the directory path below the components folder leading to the
00024  *  component. The next element ("handler-class" in this example) will be mapped to a
00025  *  component class file by "camelizing" the string to "HandlerClassHandler" and adding
00026  *  ".inc.php" to the end. The "operation-name" is transformed to "operationName"
00027  *  and represents the name of the handler method to be called. Finally "arg1", "arg2",
00028  *  etc. are parameters to be passed along to the handler method.
00029  *
00030  *  For servers with path info disabled the component URL looks like this:
00031  *
00032  *       .../index.php?component=path.to.handler-class&op=operation-name&arg1=...&arg2=...
00033  *
00034  *  The router will sanitize the request URL to a certain amount to make sure that
00035  *  random code inclusions are prevented. User authorization and parameter validation
00036  *  are however not the router's concern. These must be implemented on handler level.
00037  *
00038  *  NB: Component and operation names may only contain a-z, 0-9 and hyphens. Numbers
00039  *  are not allowed at the beginning of a name or after a hyphen.
00040  *
00041  *  NB: Component handlers must implement an initialize() method that will be called
00042  *  before the request is routed. The initialization method must enforce authorization
00043  *  and request validation.
00044  */

00399     * Try to retrieve a (non-validated) array with the service
00400     * endpoint parts from the request. See the classdoc for the
00401     * URL patterns supported here.
00402     * @param $request PKPRequest the request to be routed
00403     * @return array an array of (non-validated) service endpoint
00404     *  parts or null if the request is not an RPC request.
00405     */
00406    function _retrieveServiceEndpointParts(&$request) {
00407       // URL pattern depends on whether the server has path info
00408       // enabled or not. See classdoc for details.
00409       if ($request->isPathInfoEnabled()) {
00410          if (!isset($_SERVER['PATH_INFO'])) return null;
00411 
00412          $pathInfoParts = explode('/', trim($_SERVER['PATH_INFO'], '/'));
00413 
00414          // We expect at least the context + the component
00415          // router marker + 3 component parts (path, handler, operation)
00416          $application = $this->getApplication();
00417          $contextDepth = $application->getContextDepth();
00418          if (count($pathInfoParts) < $contextDepth + 4) {
00419             // This path info is too short to be an RPC request
00420             return null;
00421          }
00422 
00423          // Check the component router marker
00424          if ($pathInfoParts[$contextDepth] != COMPONENT_ROUTER_PATHINFO_MARKER) {
00425             // This is not an RPC request
00426             return null;
00427          }
00428 
00429          // Remove context and component marker from the array
00430          $rpcServiceEndpointParts = array_slice($pathInfoParts, $contextDepth + 1);
00431       } else {
00432          $componentParameter = $request->getUserVar(COMPONENT_ROUTER_PARAMETER_MARKER);
00433          $operationParameter = $request->getUserVar('op');
00434          if (is_null($componentParameter) || is_null($operationParameter)) {
00435             // This is not an RPC request
00436             return null;
00437          }
00438 
00439          // Expand the router parameter
00440          $rpcServiceEndpointParts = explode('.', $componentParameter);
00441 
00442          // Add the operation
00443          array_push($rpcServiceEndpointParts, $operationParameter);
00444       }
00445 
00446       return $rpcServiceEndpointParts;
00447    }

00470       foreach($rpcServiceEndpointParts as $key => $rpcServiceEndpointPart) {
