00128       if ($status == STATUS_ARCHIVED || $status == STATUS_PUBLISHED ||
00129           $status == STATUS_DECLINED) return $status;
00130 
00131       // The submission is STATUS_QUEUED or the author's submission was STATUS_INCOMPLETE.
00132       if ($this->getSubmissionProgress()) return (STATUS_INCOMPLETE);
00133 
00134       // The submission is STATUS_QUEUED. Find out where it's queued.
00135       $editAssignments = $this->getEditAssignments();
00136       if (empty($editAssignments))
00137          return (STATUS_QUEUED_UNASSIGNED);
00138

00140       $decision = array_pop($decisions);
00141       if (!empty($decision)) {
00142          $latestDecision = array_pop($decision);
00143          if ($latestDecision['decision'] == SUBMISSION_EDITOR_DECISION_ACCEPT) {
00144             return STATUS_QUEUED_EDITING;
00145          }
00146       }
00147       return STATUS_QUEUED_REVIEW;
