00014  * @brief Basic router class that has functionality common to all routers.
00015  *
00016  * NB: All handlers provide the common basic workflow. The router
00017  * calls the following methods in the given order.
00018  * 1) constructor:
00019  *       Handlers should establish a mapping of remote
00020  *       operations to roles that may access them. They do
00021  *       so by calling PKPHandler::addRoleAssignment().
00022  * 2) authorize():
00023  *       Authorizes the request, among other things based
00024  *       on the result of the role assignment created
00025  *       during object instantiation. If authorization fails
00026  *       then die with a fatal error or execute the "call-
00027  *       on-deny" advice if one has been defined in the
00028  *       authorization policy that denied access.
00029  * 3) validate():
00030  *       Let the handler execute non-fatal data integrity
00031  *       checks (FIXME: currently only for component handlers).
00032  *       Please make sure that data integrity checks that can
00033  *       lead to denial of access are being executed in the
00034  *       authorize() step via authorization policies and not
00035  *       here.
00036  * 4) initialize():
00037  *       Let the handler initialize its internal state based
00038  *       on authorized and valid data. Authorization and integrity
00039  *       checks should be kept out of here to get a clear separation
00040  *       of concerns.
00041  * 5) execution:
00042  *       Executes the requested handler operation. The mapping
00043  *       of requests to operations depends on the router
00044  *       implementation (see the class doc of specific router
00045  *       implementations for more details).
00046  * 6) client response:
00047  *       Handlers should return a string value that will then be
00048  *       returned to the client as a response. Handler operations
00049  *       should not output the response directly to the client so
00050  *       that we can run filter operations on the output if required.
00051  *       Outputting text from handler operations to the client
00052  *       is possible but deprecated.
00053  */

00338    /**
00339     * This is the method that implements the basic
00340     * life-cycle of a handler request:
00341     * 1) authorization
00342     * 2) validation
00343     * 3) initialization
00344     * 4) execution
00345     * 5) client response
00346     *
00347     * @param $serviceEndpoint callable the handler operation
00348     * @param $request PKPRequest
00349     * @param $args array
00350     * @param $validate boolean whether or not to execute the
00351     *  validation step.
00352     */
00353    function _authorizeInitializeAndCallRequest(&$serviceEndpoint, &$request, &$args, $validate = true) {
00354       assert(is_callable($serviceEndpoint));
00355 
00356       // Pass the dispatcher to the handler.
00357       $serviceEndpoint[0]->setDispatcher($this->getDispatcher());
00358 
00359       // Authorize the request.
00360       $roleAssignments = $serviceEndpoint[0]->getRoleAssignments();
00361       assert(is_array($roleAssignments));
00362       if ($serviceEndpoint[0]->authorize($request, $args, $roleAssignments)) {
00363          // Execute class-wide data integrity checks.
00364          if ($validate) $serviceEndpoint[0]->validate($request, $args);
00365 
00366          // Let the handler initialize itself.
00367          $serviceEndpoint[0]->initialize($request, $args);
00368 
00369          // Call the service endpoint.
00370          $result = call_user_func($serviceEndpoint, $args, $request);
00371       } else {
00372          // Authorization failed - try to retrieve a user
00373          // message.
00374          $authorizationMessage = $serviceEndpoint[0]->getLastAuthorizationMessage();
00375 
00376          // Set a generic authorization message if no
00377          // specific authorization message was set.
00378          if (is_null($authorizationMessage)) $authorizationMessage = 'user.authorization.accessDenied';
00379 
00380          // Handle the authorization failure.
00381          $result = $this->handleAuthorizationFailure($request, $authorizationMessage);
00382       }
00383 
00384       // Return the result of the operation to the client.
00385       if (is_string($result)) echo $result;
00386    }

00389     * Canonicalizes the new context.
00390     *
00391     * A new context can be given as a scalar. In this case only the
00392     * first context will be replaced. If the context depth of the
00393     * current application is higher than one than the context can also
00394     * be given as an array if more than the first context should
00395     * be replaced. We therefore canonicalize the new context to an array.
00396     *
00397     * When all entries are of the form 'contextName' => null or if

00400     * @param $newContext the raw context array
00401     * @return array the canonicalized context array

00404       // Create an empty array in case no new context was given.
00405       if (is_null($newContext)) $newContext = array();
00406 
00407       // If we got the new context as a scalar then transform
00408       // it into an array.
00409       if (is_scalar($newContext)) $newContext = array($newContext);
00410 
00411       // Check whether any new context has been provided.
00412       // If not then return an empty array.
00413       $newContextProvided = false;
00414       foreach($newContext as $contextElement) {
00415          if(isset($contextElement)) $newContextProvided = true;
00416       }
00417       if (!$newContextProvided) $newContext = array();

00423     * Build the base URL and add the context part of the URL.
00424     *
00425     * The new URL will be based on the current request's context
00426     * if no new context is given.
00427     *
00428     * The base URL for a given primary context can be overridden
00429     * in the config file using the 'base_url[context]' syntax in the
00430     * config file's 'general' section.
00431     *
00432     * @param $request PKPRequest the request to be routed
00433     * @param $newContext mixed (optional) context that differs from

00435     * @return array An array consisting of the base url as the first

00440 
00441       // Retrieve the context list.
00442       $contextList = $this->_contextList;
00443 
00444       // Determine URL context
00445       $context = array();

00447          if ($pathInfoEnabled) {
00448             $contextParameter = '';
00449          } else {
00450             $contextParameter = $contextName.'=';
00451          }
00452 
00453          $newContextValue = array_shift($newContext);
00454          if (isset($newContextValue)) {
00455             // A new context has been set so use it.
00456             $contextValue = rawurlencode($newContextValue);

00458             // No new context has been set so determine
00459             // the current request's context
00460             $contextObject =& $this->getContextByName($request, $contextName);
00461             if ($contextObject) $contextValue = $contextObject->getPath();

00463 
00464          }
00465

00467          if ($contextKey == 0) {

00517    /**
00518     * Creates a valid URL from parts.
00519     * @param $baseUrl string the protocol, domain and initial path/parameters, no anchors allowed here
00520     * @param $pathInfoArray array strings to be concatenated as path info
00521     * @param $queryParametersArray array strings to be concatenated as query string
00522     * @param $anchor string an additional anchor
00523     * @param $escape boolean whether to escape ampersands
00524     * @return string the URL
00525     */
00526    function _urlFromParts($baseUrl, $pathInfoArray = array(), $queryParametersArray = array(), $anchor = '', $escape = false) {
00527       // Parse the base url
00528       $baseUrlParts = parse_url($baseUrl);
00529       assert(isset($baseUrlParts['scheme']) && isset($baseUrlParts['host']) && !isset($baseUrlParts['fragment']));
00530 
00531       // Reconstruct the base url without path and query
00532       $baseUrl = $baseUrlParts['scheme'].'://';
00533       if (isset($baseUrlParts['user'])) {
00534          $baseUrl .= $baseUrlParts['user'];
00535          if (isset($baseUrlParts['pass'])) {
00536             $baseUrl .= ':'.$baseUrlParts['pass'];
00537          }
00538          $baseUrl .= '@';
00539       }
00540       $baseUrl .= $baseUrlParts['host'];
00541       if (isset($baseUrlParts['port'])) $baseUrl .= ':'.$baseUrlParts['port'];
00542       $baseUrl .= '/';
00543 
00544       // Add path info from the base URL
00545       // to the path info array (if any).
00546       if (isset($baseUrlParts['path'])) {
00547          $pathInfoArray = array_merge(explode('/', trim($baseUrlParts['path'], '/')), $pathInfoArray);
00548       }
00549 
00550       // Add query parameters from the base URL
00551       // to the query parameter array (if any).
00552       if (isset($baseUrlParts['query'])) {
00553          $queryParametersArray = array_merge(explode('&', $baseUrlParts['query']), $queryParametersArray);
00554       }
00555 
00556       // Expand path info
00557       $pathInfo = implode('/', $pathInfoArray);
00558 
00559       // Expand query parameters
00560       $amp = ($escape ? '&amp;' : '&');
00561       $queryParameters = implode($amp, $queryParametersArray);
00562       $queryParameters = (empty($queryParameters) ? '' : '?'.$queryParameters);
00563 
00564       // Assemble and return the final URL
00565       return $baseUrl.$pathInfo.$queryParameters.$anchor;
00566    }
