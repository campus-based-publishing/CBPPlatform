00068    function parseTypeName($typeName) {
00069       // We expect a validation strategy and an optional validation argument
00070       $typeNameParts = explode('(', $typeName);
00071       switch (count($typeNameParts)) {
00072          case 1:

00074             $validationStrategy = $typeName;

00084                   && $validationStrategy != XML_TYPE_DESCRIPTION_VALIDATE_RELAX_NG) return false;
