00015  * The filter DAO acts as a filter registry. It allows filter providers to
00016  * register transformations and filter consumers to identify available
00017  * transformations that convert a given input type into a required output type.
00018  *
00019  * Transformations are defined as a combination of a filter class and a pair of
00020  * input/output type specifications supported by that filter implementation.
00021  *
00022  * The following additional conditions apply:
00023  * 1) A single filter may implement several transformations, i.e. distinct
00024  *    combinations of input and output types. Therefore the filter DAO
00025  *    must be able to check all available transformations against a given input
00026  *    type and an expected output type and select those filters instances that
00027  *    support compatible transformations.
00028  * 2) The DAO must take care to only select such transformations that are
00029  *    supported by the current runtime environment.
00030  * 3) The DAO implementation must be fast and memory efficient.
00031  */

00147     * @return array all compatible filter instances (transformations).
00148     */
00149    function &getCompatibleObjects($inputSample, $outputSample, $contextId = 0, $getTemplates = false, $rehash = false) {
00150       static $filterHash = array();
00151       static $typeDescriptionFactory = null;
00152       static $typeDescriptionCache = array();
00153       static $filterInstanceCache = array();
00154 
00155       // Instantiate the type description factory
00156       if (is_null($typeDescriptionFactory)) {
00157          $typeDescriptionFactory =& TypeDescriptionFactory::getInstance();
00158       }
00159 
00160       // 1) Hash all available transformations by input
00161       //    and output type.
00162       $hashId = $contextId.'-'.($getTemplates?'1':'0');
00163       if (!isset($filterHash[$hashId]) || $rehash) {
00164          $result =& $this->retrieve(
00165             'SELECT * FROM filters'.
00166             ' WHERE '.($getTemplates ? 'is_template = 1' : 'is_template = 0').
00167             ' AND context_id in (0, ?) AND parent_filter_id = 0', (integer)$contextId);
00168          foreach($result->GetAssoc() as $filterRow) {
00169             $filterHash[$hashId][$filterRow['input_type']][$filterRow['output_type']][] = $filterRow;
00170          }
00171 
00172          // Set an empty array if no filters were found at all.
00173          if (!isset($filterHash[$hashId])) $filterHash[$hashId] = array();
00174       }
00175 
00176       // 2) Check the input sample against all input types.
00177       $intermediateCandidates = array();
00178       foreach($filterHash[$hashId] as $inputType => $outputHash) {
00179          // Instantiate the type description if not yet done
00180          // before.
00181          if (!isset($typeDescriptionCache[$inputType])) {
00182             $typeDescriptionCache[$inputType] =& $typeDescriptionFactory->instantiateTypeDescription($inputType);
00183          }
00184 
00185          // 3) Whenever an input type matches, hash all filters
00186          //    with this input type by output type.
00187          if ($typeDescriptionCache[$inputType]->checkType($inputSample)) {
00188             foreach($outputHash as $outputType => $filterRows) {
00189                if (!isset($intermediateCandidates[$outputType])) {
00190                   $intermediateCandidates[$outputType] = $filterRows;
00191                } else {
00192                   $intermediateCandidates[$outputType] = array_merge($intermediateCandidates[$outputType], $filterRows);
00193                }
00194             }
00195          }
00196       }
00197 
00198       // 4) Check the output sample against all hashed output
00199       //    types.
00200       $matchingFilterRows = array();
00201       foreach ($intermediateCandidates as $outputType => $filterRows) {
00202          // Instantiate the type description if not yet done
00203          // before.
00204          if (!isset($typeDescriptionCache[$outputType])) {
00205             $typeDescriptionCache[$outputType] =& $typeDescriptionFactory->instantiateTypeDescription($outputType);
00206          }
00207 
00208          // 5) Whenever an output type matches, add all filters
00209          //    with this output type to the result set.
00210          if ($typeDescriptionCache[$outputType]->checkType($outputSample)) {
00211             $matchingFilterRows = array_merge($matchingFilterRows, $filterRows);
00212          }
00213       }
00214 
00215       // 6) Instantiate and return all transformations in the
00216       //    result set that comply with the current runtime
00217       //    environment.
00218       $matchingFilters = array();
00219       $runTimeRequirementNotMet = -1;
00220       foreach($matchingFilterRows as $matchingFilterRow) {
00221          if (!isset($filterInstanceCache[$matchingFilterRow['filter_id']])) {
00222             $filterInstance =& $this->_fromRow($matchingFilterRow);
00223             if ($filterInstance->isCompatibleWithRuntimeEnvironment()) {
00224                $filterInstanceCache[$matchingFilterRow['filter_id']] =& $filterInstance;
00225             } else {
00226                $filterInstanceCache[$matchingFilterRow['filter_id']] = $runTimeRequirementNotMet;
00227             }
00228             unset($filterInstance);
00229          }
00230          if ($filterInstanceCache[$matchingFilterRow['filter_id']] !== $runTimeRequirementNotMet) {
00231             $matchingFilters[$matchingFilterRow['filter_id']] = $filterInstanceCache[$matchingFilterRow['filter_id']];
00232          }
00233       }
00234 
00235       return $matchingFilters;
00236    }

00422    /**
00423     * Populate the sub-filters (if any) for the
00424     * given parent filter.
00425     * @param $parentFilter Filter
00426     */
00427    function _populateSubFilters(&$parentFilter) {
00428       if (!is_a($parentFilter, 'CompositeFilter')) {
00429          // Nothing to do. Only composite filters
00430          // can have sub-filters.
00431          return;
00432       }
00433 
00434       // Retrieve the sub-filters from the database.
00435       $parentFilterId = $parentFilter->getId();
00436       $result =& $this->retrieve(
00437             'SELECT * FROM filters WHERE parent_filter_id = ? ORDER BY seq', $parentFilterId);
00438       $daoResultFactory = new DAOResultFactory($result, $this, '_fromRow', array('filter_id'));
00439 
00440       // Add sub-filters.
00441       while (!$daoResultFactory->eof()) {
00442          // Retrieve the sub filter.
00443          // NB: This recursively loads sub-filters
00444          // of this filter via _fromRow().
00445          $subFilter =& $daoResultFactory->next();
00446 
00447          // Add the sub-filter to the filter list
00448          // of its parent filter.
00449          $parentFilter->addFilter($subFilter);
00450          unset($subFilter);
00451       }
00452    }
