00310       // Separate the element name from the attributes.
00311       $elementPlusAttributes = explode('[', $nodePath);
00312       $element = $elementPlusAttributes[0];
00313       assert(!empty($element));
00314 
00315       // Create the element.
00316       $newNode =& XMLCustomWriter::createElement($doc, $element);
00317 
00318       // Add attributes.
00319       if (count($elementPlusAttributes) == 2) {
00320          // Separate the attribute key/value pairs.
00321          $unparsedAttributes = explode('@', rtrim(ltrim($elementPlusAttributes[1], '@'), ']'));
00322          foreach($unparsedAttributes as $unparsedAttribute) {
00323             // Split attribute expressions into key and value.
00324             list($attributeName, $attributeValue) = explode('=', rtrim($unparsedAttribute, ' '));
00325             $attributeValue = trim($attributeValue, '"');
00326             XMLCustomWriter::setAttribute($newNode, $attributeName, $attributeValue);
00327          }
00328       }
00329 
00330       // Insert a text node if we got a value for it.
00331       if (!is_null($value)) {
00332          $textNode =& XMLCustomWriter::createTextNode($doc, $value);
00333          XMLCustomWriter::appendChild($newNode, $textNode);
