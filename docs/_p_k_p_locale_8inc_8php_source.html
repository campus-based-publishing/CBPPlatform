00356    /**
00357     * Translate an ISO639-2b compatible 3-letter string
00358     * into the PKP locale identifier.
00359     *
00360     * This can be ambiguous if several locales are defined
00361     * for the same language. In this case we'll use the
00362     * primary locale to disambiguate.
00363     *
00364     * If that still doesn't determine a unique locale then
00365     * we'll choose the first locale found.
00366     *
00367     * @param $iso3letter string
00368     * @return string
00369     */
00370    function getLocaleFrom3LetterIso($iso3Letter) {
00371       assert(strlen($iso3Letter) == 3);
00372       $primaryLocale = Locale::getPrimaryLocale();
00373 
00374       $localeCandidates = array();
00375       $locales =& Locale::_getAllLocalesCacheContent();
00376       foreach($locales as $locale => $localeData) {
00377          assert(isset($localeData['iso639-2b']));
00378          if ($localeData['iso639-2b'] == $iso3Letter) {
00379             if ($locale == $primaryLocale) {
00380                // In case of ambiguity the primary locale
00381                // overrides all other options so we're done.
00382                return $primaryLocale;
00383             }
00384             $localeCandidates[] = $locale;
00385          }
00386       }
00387 
00388       // Return null if we found no candidate locale.
00389       if (empty($localeCandidates)) return null;
00390 
00391       if (count($localeCandidates) > 1) {
00392          // Check whether one of the candidate locales
00393          // is a supported locale. If so choose the first
00394          // supported locale.
00395          $supportedLocales = Locale::getSupportedLocales();
00396          foreach($supportedLocales as $supportedLocale => $localeName) {
00397             if (in_array($supportedLocale, $localeCandidates)) return $supportedLocale;
00398          }
00399       }
00400 
00401       // If there is only one candidate (or if we were
00402       // unable to disambiguate) then return the unique
00403       // (first) candidate found.
00404       return array_shift($localeCandidates);
00405    }
