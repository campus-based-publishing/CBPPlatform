00058    /**
00059     * Remove an author.
00060     * @param $authorId ID of the author to remove
00061     * @return boolean author was removed
00062     */
00063    function removeAuthor($authorId) {
00064       $found = false;
00065 
00066       if ($authorId != 0) {
00067          $authors = array();
00068          for ($i=0, $count=count($this->authors); $i < $count; $i++) {
00069             if ($this->authors[$i]->getId() == $authorId) {
00070                array_push($this->removedAuthors, $authorId);
00071                $found = true;
00072             } else {
00073                array_push($authors, $this->authors[$i]);
00074             }
00075          }
00076          $this->authors = $authors;
00077       }
00078       return $found;
00079    }
