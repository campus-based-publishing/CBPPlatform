00057             $payment->setRequestUrl(Request::url(null, 'user') );

00059          case PAYMENT_TYPE_PURCHASE_SUBSCRIPTION:
00060          case PAYMENT_TYPE_RENEW_SUBSCRIPTION:
00061             $payment->setRequestUrl(Request::url(null, 'user', 'subscriptions') );
00062             break;
00063          case PAYMENT_TYPE_DONATION:
00064             $payment->setRequestUrl(Request::url(null, 'donations', 'thankYou') );
00065             break;
00066          case PAYMENT_TYPE_FASTTRACK:
00067          case PAYMENT_TYPE_PUBLICATION:
00068          case PAYMENT_TYPE_SUBMISSION:
00069             $authorSubmissionDao =& DAORegistry::getDAO('AuthorSubmissionDAO');
00070             $authorSubmission =& $authorSubmissionDao->getAuthorSubmission($assocId);
00071             if ($authorSubmission->getSubmissionProgress()!=0) {
00072                $payment->setRequestUrl(Request::url(null, 'author', 'submit', $authorSubmission->getSubmissionProgress(), array('articleId' => $assocId)));
00073             } else {
00074                $payment->setRequestUrl(Request::url(null, 'author') );
00075             }
00076             break;

00078             // something went wrong. crud.
00079             break;
