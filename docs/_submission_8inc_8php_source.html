00045    /**
00046     * Get a piece of data for this object, localized to the current
00047     * locale if possible.
00048     * @param $key string
00049     * @return mixed
00050     */
00051    function &getLocalizedData($key) {
00052       $localePrecedence = array(Locale::getLocale(), $this->getLocale());
00053       foreach ($localePrecedence as $locale) {
00054          $value =& $this->getData($key, $locale);
00055          if (!empty($value)) return $value;
00056          unset($value);
00057       }
00058 
00059       // Fallback: Get the first available piece of data.
00060       $data =& $this->getData($key, null);
00061       if (!empty($data)) return $data[array_shift(array_keys($data))];
00062 
00063       // No data available; return null.
00064       unset($data);
00065       $data = null;
00066       return $data;
00067    }
