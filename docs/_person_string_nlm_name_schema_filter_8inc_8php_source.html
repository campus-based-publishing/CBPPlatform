00151     * Converts a string with multiple persons

00161    function &_parsePersonsString($personsString, $title, $degrees) {
00162       // Check for 'et al'.
00163       $personsStringBeforeEtal = String::strlen($personsString);
00164       $personsString = String::regexp_replace('/et ?al$/', '', $personsString);
00165       $etAl = ($personsStringBeforeEtal == String::strlen($personsString) ? false : true);
00166 
00167       // Remove punctuation.
00168       $personsString = trim($personsString, ':;, ');
00169 
00170       // Cut the authors string into pieces.
00171       $personStrings = String::iterativeExplode(array(':', ';'), $personsString);
00172 
00173       // If we did not have success with simple patterns then try more complex
00174       // patterns to tokenize multiple-person strings.
00175       if (count($personStrings) == 1) {
00176          // The first pattern must match the whole string, the second is used
00177          // to extract names.
00178          $complexPersonsPatterns = array(
00179             // APA author style unfortunately uses commas to separate surnames
00180             // as well as pre-names which makes it much more difficult to split
00181             // correctly. We allow for non-APA whitespacing and punctuation.
00182             array(
00183                '/^((([^ \t\n\r\f\v,.&]{2,}\s*)+,\s*([A-Z]\.\s*)+),\s*)+(\&|\.\s\.\s\.)\s*([^ \t\n\r\f\v,.&]{2,}\s*,\s*([A-Z]\.\s*)+)$/i',
00184                '/(?:[^ \t\n\r\f\v,.&]{2,}\s*)+,\s*(?:[A-Z]\.\s*)+/i'
00185             ),
00186             // Try to cut by comma if the pieces contain more
00187             // than one word to avoid splitting between last name and
00188             // first name. This captures APA editor and Vancouver
00189             // author styles for example.
00190             array(
00191                '/^((([^ \t\n\r\f\v,&]+\s+)+[^ \t\n\r\f\v,&]+\s*)[,&]\s*)+(([^ \t\n\r\f\v,&]+\s+)+[^ \t\n\r\f\v,&]+)/i',
00192                '/(?:(?:[^ \t\n\r\f\v,&.]+|[^ \t\n\r\f\v,&]{2,})\s+)+(?:[^ \t\n\r\f\v,&.]+|[^ \t\n\r\f\v,&]{2,})/i'
00193             )
00194          );
00195          $matched = false;
00196          foreach($complexPersonsPatterns as $complexPersonsPattern) {
00197             // Break at the first pattern that matches.
00198             if ($matched = String::regexp_match($complexPersonsPattern[0], $personsString)) {
00199                // Retrieve names.
00200                $success = String::regexp_match_all($complexPersonsPattern[1], $personsString, $personStrings);
00201                assert($success && count($personStrings) == 1);
00202                $personStrings = $personStrings[0];
00203                break;
00204             }
00205          }
00206 
00207          if (!$matched) {
00208             // If nothing matches then try to parse as a single person.
00209             $personStrings = array($personsString);
00210          }
00211       }
00212 
00213       // Parse persons.
00214       $persons = array();
00215       foreach ($personStrings as $personString) {
00216          $persons[] =& $this->_parsePersonString($personString, $title, $degrees);
00217       }
00218 
00219       // Add et-al string.
00220       if ($etAl) {
00221          $persons[] = PERSON_STRING_FILTER_ETAL;
00222       }
00223 
00224       return $persons;
00225    }
