00118       // This is a two-dimensional array that with the score as key and
00119       // the scored citations as values.
00120       $scoredCitations = array();

00123       // scored. This is to make sure that we don't downgrade
00124       // our results.
00125       $citationOptions = $input;
00126       $citationOptions[] =& $this->getOriginalDescription();
00127 
00128       // Iterate over the incoming NLM citation descriptions
00129       foreach ($citationOptions as $citationOption) {
00130          // If the publication type is not set, take a guess
00131          if (!$citationOption->hasStatement('[@publication-type]')) {
00132             $guessedPublicationType = $this->_guessPublicationType($citationOption);
00133             if (!is_null($guessedPublicationType)) {
00134                $citationOption->addStatement('[@publication-type]', $guessedPublicationType);
00135             }
00136          }
00137 
00138          // Calculate the score for this filtered citation
00139          $confidenceScore = $this->_filterConfidenceScore($citationOption);
00140 
00141          // Save the filtered result hashed by its confidence score.
00142          // We save them as a sub-array in case several citations
00143          // receive the same confidence score.
00144          if (!isset($scoredCitations[$confidenceScore])) {
00145             $scoredCitations[$confidenceScore] = array();
00146          }
00147          $scoredCitations[$confidenceScore][] =& $citationOption;
00148          unset ($citationOption);
00149       }
00150 
00151       // Get a single set of "best" values for the citation description
00152       // and set them in a new citation object. Don't accept results that
00153       // don't stem from citations that got at least 50% of the original
00154       // text right.
00155       $citation =& $this->_guessValues($scoredCitations, 50);
00156       return $citation;

00200             // Record the hit
00201             $hitCounters[$currentProbablePublicationType]++;
00202 
00203             // Is this currently the highest counter value?
00204             if ($hitCounters[$currentProbablePublicationType] > $highestCounterValue) {
00205                // This is the highest value
00206                $highestCounterValue = $hitCounters[$currentProbablePublicationType];
00207                $probablePublicationType = $currentProbablePublicationType;
00208             } elseif ($hitCounters[$currentProbablePublicationType] == $highestCounterValue) {
00209                // There are two counters with the same value, so no unique result
00210                $probablePublicationType = null;
00211             }
00212          }
00213       }
00214

00220     * Derive a confidence score calculated as the similarity of the
00221     * original raw citation and the citation text generated from the
00222     * citation description.
00223     * @param $metadataDescription MetadataDescription

00227       // Retrieve the original plain text citation.
00228       $originalCitation = $this->getOriginalRawCitation();
00229 
00230       // Generate the formatted citation output from the description.
00231       $citationOutputFilter =& $this->getCitationOutputFilter();
00232       $generatedCitation = $citationOutputFilter->execute($metadataDescription);
00233 
00234       // Strip formatting and the Google Scholar tag so that we get a plain
00235       // text string that is comparable with the raw citation.
00236       $generatedCitation = trim(str_replace(GOOGLE_SCHOLAR_TAG, '', strip_tags($generatedCitation)));
00237 
00238       // Compare the original to the generated citation.
00239       $citationDiff = String::diff($originalCitation, $generatedCitation);
00240 
00241       // Calculate similarity as the number of deleted characters in relation to the
00242       // number of characters in the original citation. This intentionally excludes
00243       // additions as these can represent useful data like a DOI or an external link.
00244       $deletedCharacters = 0;
00245       foreach($citationDiff as $diffPart) {
00246          // Identify deletions.
00247          if (key($diffPart) == -1) {
00248             $deletedCharacters += String::strlen(current($diffPart));
00249          }
00250       }
00251       $originalCharacters = String::strlen($originalCitation);
00252       $partOfCommonCharacters = ($originalCharacters-$deletedCharacters) / $originalCharacters;
00253 
00254       $filterConfidenceScore = (integer)round(min($partOfCommonCharacters*100, 100));
00255       return $filterConfidenceScore;
00256    }

00258    /**
00259     * Take an array of citation parse/lookup results and derive a citation
00260     * with one "best" set of values.
00261     *
00262     * We determine the best values within the citations that have a score above
00263     * the given threshold. Citations with a score below the threshold will be
00264     * ignored.
00265     *
00266     * For these citations we count the frequency of values per meta-data property.
00267     * The most frequent value will be chosen as "best" value.
00268     *
00269     * If two values have the same frequency then decide based on the score. If
00270     * this is still ambivalent then return the first of the remaining values.
00271     *
00272     * @param $scoredCitations
00273     * @param $scoreThreshold integer a number between 0 (=no threshold) and 100
00274     * @return Citation one citation with the "best" values set
00275     */

00277       assert($scoreThreshold >= 0 && $scoreThreshold <= 100);
00278 
00279       // Create the target citation description.
00280       $targetDescription = new MetadataDescription('lib.pkp.classes.metadata.nlm.NlmCitationSchema', ASSOC_TYPE_CITATION);
00281 
00282       // Step 1: List all values and max scores that have been identified for a given element
00283       //         but only include values from results above a given scoring threshold
00284 
00285       // Initialize variables for the first step.
00286       $valuesByPropertyName = array();
00287       $maxScoresByPropertyNameAndValue = array();
00288 
00289       // Sort the scored citations by score with the highest score first.
00290       krsort($scoredCitations);
00291       foreach ($scoredCitations as $currentScore => $citationsForCurrentScore) {
00292          // Check whether the current score is below the threshold, if so
00293          // stop the loop. We've sorted our citations by score so the remaining
00294          // citations all have scores below the threshold and we can forget
00295          // about them.
00296          if ($currentScore < $scoreThreshold) {
00297             break;
00298          }
00299 
00300          foreach($citationsForCurrentScore as $citationForCurrentScore) {
00301             $statements = $citationForCurrentScore->getStatements();
00302 
00303             // Add the property values and scores of this citation
00304             // to the overall property lists
00305             foreach($statements as $propertyName => $value) {
00306                // Initialize sub-arrays if necessary
00307                if (!isset($valuesByPropertyName[$propertyName])) {
00308                   $valuesByPropertyName[$propertyName] = array();
00309                }
00310                if (!isset($maxScoresByPropertyNameAndValue[$propertyName])) {
00311                   $maxScoresByPropertyNameAndValue[$propertyName] = array();
00312                }
00313 
00314                // Add the value for the given property, as we want to count
00315                // value frequencies later, we explicitly allow duplicates.
00316                $serializedValue = serialize($value);
00317                $valuesByPropertyName[$propertyName][] = $serializedValue;
00318 
00319                // As we have ordered our citations descending by score, the
00320                // first score found for a value is also the maximum score.
00321                if (!isset($maxScoresByPropertyNameAndValue[$propertyName][$serializedValue])) {
00322                   $maxScoresByPropertyNameAndValue[$propertyName][$serializedValue] = $currentScore;
00323                }
00324             }
00325          }
00326       }
00327 
00328       // Step 2: Find out the values that occur most frequently for each element
00329       //         and order these by score.
00330 
00331       foreach($valuesByPropertyName as $propertyName => $values) {
00332          // Count the occurrences of each value within the given element
00333          $valueFrequencies = array_count_values($values);
00334 
00335          // Order the most frequent values to the beginning of the array
00336          arsort($valueFrequencies);
00337 
00338          // Get the most frequent values (may be several if there are more than one
00339          // with the same frequency).
00340          $scoresOfMostFrequentValues = array();
00341          $previousValueFrequency = 0;
00342          foreach($valueFrequencies as $value => $valueFrequency) {
00343             // Only extract the most frequent values, jump out of the
00344             // loop when less frequent values start.
00345             if ($previousValueFrequency > $valueFrequency) break;
00346             $previousValueFrequency = $valueFrequency;
00347 
00348             $scoresOfMostFrequentValues[$value] =
00349                   $maxScoresByPropertyNameAndValue[$propertyName][$value];
00350          }
00351 
00352          // Now we can order the most frequent values by score, starting
00353          // with the highest score.
00354          arsort($scoresOfMostFrequentValues);
00355 
00356          // Now get the first key which represents the value with the
00357          // highest frequency and the highest score.
00358          reset($scoresOfMostFrequentValues);
00359          $bestValue = unserialize(key($scoresOfMostFrequentValues));
00360 
00361          // Set the found "best" element value in the result citation.
00362          $statements = array($propertyName => $bestValue);
00363          $success = $targetDescription->setStatements($statements);
00364          assert($success);
00365       }
