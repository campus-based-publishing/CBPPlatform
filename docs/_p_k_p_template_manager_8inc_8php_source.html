00934    /**
00935     * Helper function: Recursive function called by _removeTags
00936     * Removes tags from the back of the string and keeps a record of their position from the back
00937     * @author Matt Crider
00938     * @param string
00939     * @param int loc Keeps track of position from the back of original string
00940     * @param array
00941     * @param int
00942     * @return string
00943     */
00944    function _removeTagsAuxReverse($string, $loc, &$tags, $length) {
00945       $backLoc = String::strlen($string)-1;
00946       if($backLoc >= 0 && $length > 0) {
00947          $length--;
00948          if(String::substr($string, $backLoc, 1) == '>') {
00949             $tag = '>';
00950             $openBrack = 1;
00951             while (String::substr($string, $backLoc-$openBrack, 1) != '<') {
00952                $tag = String::substr($string, $backLoc-$openBrack, 1) . $tag;
00953                $openBrack++;
00954             }
00955             $tag = '<' . $tag;
00956             $openBrack++;
00957 
00958             $tags[] = array($tag, $loc);
00959             return $this->_removeTagsAuxReverse(String::substr($string, 0, -$openBrack), $loc+$openBrack, $tags, $length);
00960          }
00961          return $this->_removeTagsAuxReverse(String::substr($string, 0, -1), $loc+1, $tags, $length) . String::substr($string, $backLoc, 1);
00962       }
00963    }

00995    /**
00996     * Helper function: Closes all dangling XHTML tags in a string
00997     * Modified from http://milianw.de/code-snippets/close-html-tags
00998     *  by Milian Wolff <mail@milianw.de>
00999     * @param string
01000     * @return string
01001     */
01002    function _closeTags($string, $open = false){
01003       // Put all opened tags into an array
01004       String::regexp_match_all("#<([a-z]+)( .*)?(?!/)>#iU", $string, $result);
01005       $openedtags = $result[1];
01006 
01007       // Put all closed tags into an array
01008       String::regexp_match_all("#</([a-z]+)>#iU", $string, $result);
01009       $closedtags = $result[1];
01010       $len_opened = count($openedtags);
01011       $len_closed = count($closedtags);
01012       // All tags are closed
01013       if(count($closedtags) == $len_opened){
01014          return $string;
01015       }
01016 
01017       $openedtags = array_reverse($openedtags);
01018       $closedtags = array_reverse($closedtags);
01019 
01020       if ($open) {
01021          // Open tags
01022          for($i=0; $i < $len_closed; $i++) {
01023             if (!in_array($closedtags[$i],$openedtags)){
01024                $string = '<'.$closedtags[$i].'>' . $string;
01025             } else {
01026                unset($openedtags[array_search($closedtags[$i],$openedtags)]);
01027             }
01028          }
01029          return $string;
01030       } else {
01031          // Close tags
01032          for($i=0; $i < $len_opened; $i++) {
01033             if (!in_array($openedtags[$i],$closedtags)){
01034                $string .= '</'.$openedtags[$i].'>';
01035             } else {
01036                unset($closedtags[array_search($openedtags[$i],$closedtags)]);
01037             }
01038          }
01039          return $string;
01040       }
01041    }
