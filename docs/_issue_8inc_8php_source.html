00664    /**

00666     * @param $default bool labelFormat type
00667     * @param $breadcrumb bool return type of label
00668     * @param $long bool long format of label
00669     * @return string
00670     */
00671    function getIssueIdentification($default = false, $breadcrumb = false, $long = false) {
00672 
00673       if ($default) {

00675          $showNumber = 1;

00686          $showTitle = 0;

00688 
00689       $volLabel = Locale::translate('issue.vol');
00690       $numLabel = Locale::translate('issue.no');
00691 
00692       $vol = $this->getData('volume');
00693       $num = $this->getData('number');
00694       $year = $this->getData('year');
00695       $title = $this->getLocalizedData('title');
00696 
00697       $identification = '';
00698 
00699       if ($showVolume) {
00700          $identification = "$volLabel $vol";
00701       }
00702       if ($showNumber) {
00703          if (!empty($identification)) {
00704             $identification .= ", ";
00705          }
00706          $identification .= "$numLabel $num";
00707       }
00708       if ($showYear) {
00709          if (!empty($identification)) {
00710             $identification .= " ($year)";
00711          } else {
00712             $identification = "$year";
00713          }
00714       }
00715 
00716       if ($showTitle || ($long && !empty($title))) {
00717          if (!empty($identification)) {
00718             $identification .= ': ';
00719          }
00720          $identification .= "$title";
00721       }
00722 
00723       if (empty($identification)) {
00724          $identification = "$volLabel $vol, $numLabel $num ($year)";
