00266     * Validate a given input against the property specification
00267     *
00268     * The given value must validate against at least one of the

00280       // We never accept null values or arrays.
00281       if (is_null($value) || is_array($value)) return false;
00282 
00283       // Translate the locale.
00284       if (is_null($locale)) $locale = '';
00285 
00286       // MetadataProperty::getSupportedTypes() returns an ordered

00290       $allowedTypes = $this->getAllowedTypes();

00292          if (isset($allowedTypes[$testedType])) {
00293             foreach ($allowedTypes[$testedType] as $allowedTypeParam) {
00294                // Type specific validation
00295                switch ($testedType) {
00296                   case METADATA_PROPERTY_TYPE_COMPOSITE:
00297                      // Composites can either be represented by a meta-data description
00298                      // or by a string of the form AssocType:AssocId if the composite
00299                      // has already been persisted in the database.
00300                      switch(true) {
00301                         // Test for MetadataDescription format
00302                         case is_a($value, 'MetadataDescription'):
00303                            $assocType = $value->getAssocType();
00304                            break;
00305 
00306                         // Test for AssocType:AssocId format
00307                         case is_string($value):
00308                            $valueParts = explode(':', $value);
00309                            if (count($valueParts) != 2) break 2; // break the outer switch
00310                            list($assocType, $assocId) = $valueParts;
00311                            if (!(is_numeric($assocType) && is_numeric($assocId))) break 2; // break the outer switch
00312                            $assocType = (integer)$assocType;
00313                            break;
00314 
00315                         default:
00316                            // None of the allowed types
00317                            break;
00318                      }
00319 
00320                      // Check that the association type matches
00321                      // with the allowed association type (which
00322                      // is configured as an additional type parameter).
00323                      if (isset($assocType) && $assocType === $allowedTypeParam) return array(METADATA_PROPERTY_TYPE_COMPOSITE => $assocType);
00324                      break;
00325 
00326                   case METADATA_PROPERTY_TYPE_VOCABULARY:
00327                      // Interpret the type parameter of this type like this:
00328                      // symbolic[:assoc-type:assoc-id]. If no assoc type/id are
00329                      // given then we assume :0:0 to represent site-wide vocabs.
00330                      $vocabNameParts = explode(':', $allowedTypeParam);
00331                      $vocabNamePartsCount = count($vocabNameParts);

00333                         case 1:
00334                            // assume a site-wide vocabulary
00335                            $symbolic = $allowedTypeParam;
00336                            $assocType = $assocId = 0;
00337                            break;
00338 
00339                         case 3:
00340                            // assume a context-specific vocabulary
00341                            list($symbolic, $assocType, $assocId) = $vocabNameParts;
00342                            break;
00343 
00344                         default:
00345                            // Invalid configuration
00346                            assert(false);
00347                      }

00350                         // Try to translate the string value into a controlled vocab entry
00351                         $controlledVocabEntryDao =& DAORegistry::getDao('ControlledVocabEntryDAO'); /* @var $controlledVocabEntryDao ControlledVocabEntryDAO */

00354                            return array(METADATA_PROPERTY_TYPE_VOCABULARY => $allowedTypeParam);
00355                         }
00356                      }

00360                         import('lib.pkp.classes.validation.ValidatorControlledVocab');

00364                         }
00365                      }
00366 
00367                      break;
00368 
00369                   case METADATA_PROPERTY_TYPE_URI:
00370                      // Validate with the URI validator
00371                      import('lib.pkp.classes.validation.ValidatorUri');
00372                      $validator = new ValidatorUri();
