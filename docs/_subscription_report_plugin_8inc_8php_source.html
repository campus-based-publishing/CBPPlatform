00264    /**
00265     * Replace HTML "newline" tags (p, li, br) with line feeds. Strip all other tags.
00266     * @param $html String Input HTML string
00267     * @return String Text with replaced and stripped HTML tags
00268     */
00269    function _html2text($html) {
00270       $html = String::regexp_replace('/<[\/]?p>/', chr(10), $html);
00271       $html = String::regexp_replace('/<li>/', '&bull; ', $html);
00272       $html = String::regexp_replace('/<\/li>/', chr(10), $html);
00273       $html = String::regexp_replace('/<br[ ]?[\/]?>/', chr(10), $html);
00274       $html = String::html2utf(strip_tags($html));
00275       return $html;
00276    }
