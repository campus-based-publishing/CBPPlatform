00022     * Generate the export DOM tree for a given journal.
00023     * @param $doc object DOM object
00024     * @param $journal object Journal to export
00025     */
00026    function &generateJournalDom(&$doc, &$journal) {
00027       $issueDao =& DAORegistry::getDAO('IssueDAO');
00028       $sectionDao =& DAORegistry::getDAO('SectionDAO');
00029       $pubArticleDao =& DAORegistry::getDAO('PublishedArticleDAO');
00030 
00031       // Records node contains all articles, each called a record
00032       $records =& XMLCustomWriter::createElement($doc, 'records');
00033 
00034       $pubArticles =& $pubArticleDao->getPublishedArticlesByJournalId($journal->getId());
00035       while ($pubArticle =& $pubArticles->next()) {
00036          $issue =& $issueDao->getIssueById($pubArticle->getIssueId());
00037          if(!$issue) continue;
00038          $section =& $sectionDao->getSection($pubArticle->getSectionId());
00039 
00040          $articleNode =& DOAJExportDom::generateArticleDom($doc, $journal, $issue, $section, $pubArticle);
00041 
00042          XMLCustomWriter::appendChild($records, $articleNode);
00043 
00044          unset($issue, $section, $articleNode);
00045       }
00046 
00047       return $records;
00048    }

00050    /**
00051     * Generate the DOM tree for a given article.
00052     * @param $doc object DOM object
00053     * @param $journal object Journal
00054     * @param $issue object Issue
00055     * @param $section object Section
00056     * @param $article object Article
00057     */
00058    function &generateArticleDom(&$doc, &$journal, &$issue, &$section, &$article) {
00059       $root =& XMLCustomWriter::createElement($doc, 'record');
00060 
00061       /* --- Article Language --- */
00062       XMLCustomWriter::createChildWithText($doc, $root, 'language', DOAJExportDom::mapLang($article->getLanguage()), false);
00063 
00064       /* --- Publisher name (i.e. institution name) --- */
00065       XMLCustomWriter::createChildWithText($doc, $root, 'publisher', $journal->getSetting('publisherInstitution'), false);
00066 
00067       /* --- Journal's title --- */
00068       XMLCustomWriter::createChildWithText($doc, $root, 'journalTitle', $journal->getLocalizedTitle(), false);
00069 
00070       /* --- Identification Numbers --- */
00071       XMLCustomWriter::createChildWithText($doc, $root, 'issn', $journal->getSetting('printIssn'), false);
00072       XMLCustomWriter::createChildWithText($doc, $root, 'eissn', $journal->getSetting('onlineIssn'), false);
00073 
00074       /* --- Article's publication date, volume, issue, DOI --- */
00075       XMLCustomWriter::createChildWithText($doc, $root, 'publicationDate', DOAJExportDom::formatDate($issue->getDatePublished()), false);
00076 
00077       XMLCustomWriter::createChildWithText($doc, $root, 'volume',  $issue->getVolume(), false);
00078 
00079       XMLCustomWriter::createChildWithText($doc, $root, 'issue',  $issue->getNumber(), false);
00080 
00081       /** --- FirstPage / LastPage (from PubMed plugin)---
00082        * there is some ambiguity for online journals as to what
00083        * "page numbers" are; for example, some journals (eg. JMIR)
00084        * use the "e-location ID" as the "page numbers" in PubMed
00085        */
00086       $pages = $article->getPages();
00087       if (preg_match("/([0-9]+)\s*-\s*([0-9]+)/i", $pages, $matches)) {
00088          // simple pagination (eg. "pp. 3-8")
00089          XMLCustomWriter::createChildWithText($doc, $root, 'startPage', $matches[1]);
00090          XMLCustomWriter::createChildWithText($doc, $root, 'endPage', $matches[2]);
00091       } elseif (preg_match("/(e[0-9]+)/i", $pages, $matches)) {
00092          // elocation-id (eg. "e12")
00093          XMLCustomWriter::createChildWithText($doc, $root, 'startPage', $matches[1]);
00094          XMLCustomWriter::createChildWithText($doc, $root, 'endPage', $matches[1]);
00095       }
00096 
00097       XMLCustomWriter::createChildWithText($doc, $root, 'doi',  $article->getDoi(), false);
00098 
00099       /* --- Article's publication date, volume, issue, DOI --- */
00100       XMLCustomWriter::createChildWithText($doc, $root, 'publisherRecordId',  $article->getPubId(), false);
00101 
00102       XMLCustomWriter::createChildWithText($doc, $root, 'documentType',  $article->getLocalizedType(), false);
00103 
00104       /* --- Article title --- */
00105       foreach ((array) $article->getTitle(null) as $locale => $title) {
00106          if (empty($title)) continue;
00107 
00108          $titleNode =& XMLCustomWriter::createChildWithText($doc, $root, 'title', $title);
00109          if (strlen($locale) == 5) XMLCustomWriter::setAttribute($titleNode, 'language', DOAJExportDom::mapLang(String::substr($locale, 0, 2)));
00110       }
00111 
00112       /* --- Authors and affiliations --- */
00113       $authors =& XMLCustomWriter::createElement($doc, 'authors');
00114       XMLCustomWriter::appendChild($root, $authors);
00115 
00116       $affilList = DOAJExportDom::generateAffiliationsList($article->getAuthors());
00117 
00118       foreach ($article->getAuthors() as $author) {
00119          $authorNode =& DOAJExportDom::generateAuthorDom($doc, $root, $issue, $article, $author, $affilList);
00120          XMLCustomWriter::appendChild($authors, $authorNode);
00121          unset($authorNode);
00122       }
00123       
00124       if (!empty($affilList[0])) {
00125          $affils =& XMLCustomWriter::createElement($doc, 'affiliationsList');
00126          XMLCustomWriter::appendChild($root, $affils);
00127             
00128          for ($i = 0; $i < count($affilList); $i++) {
00129             $affilNode =& XMLCustomWriter::createChildWithText($doc, $affils, 'affiliationName', $affilList[$i]);
00130             XMLCustomWriter::setAttribute($affilNode, 'affiliationId', $i);
00131             unset($affilNode);
00132          }
00133       }
00134       
00135       /* --- Abstract --- */
00136       foreach ((array) $article->getAbstract(null) as $locale => $abstract) {
00137          if (empty($abstract)) continue;
00138 
00139          $abstractNode =& XMLCustomWriter::createChildWithText($doc, $root, 'abstract', $abstract);
00140          if (strlen($locale) == 5) XMLCustomWriter::setAttribute($abstractNode, 'language', DOAJExportDom::mapLang(String::substr($locale, 0, 2)));
00141       }
00142 
00143       /* --- FullText URL --- */
00144       $fullTextUrl =& XMLCustomWriter::createChildWithText($doc, $root, 'fullTextUrl', Request::url(null, 'article', 'view', $article->getId()));
00145       XMLCustomWriter::setAttribute($fullTextUrl, 'format', 'html');
00146 
00147       /* --- Keywords --- */
00148       $keywords =& XMLCustomWriter::createElement($doc, 'keywords');
00149       XMLCustomWriter::appendChild($root, $keywords);
00150 
00151       $subjects = array_map('trim', explode(';', $article->getLocalizedSubject()));
00152 
00153       foreach ($subjects as $keyword) {
00154          XMLCustomWriter::createChildWithText($doc, $keywords, 'keyword', $keyword, false);
00155       }
00156 
00157       return $root;
00158    }

00161     * Generate the author export DOM tree.
00162     * @param $doc object DOM object
00163     * @param $journal object Journal
00164     * @param $issue object Issue
00165     * @param $article object Article
00166     * @param $author object Author
00167     * @param $affilList array List of author affiliations
00168     */
00169    function &generateAuthorDom(&$doc, &$journal, &$issue, &$article, &$author, &$affilList) {
00170       $root =& XMLCustomWriter::createElement($doc, 'author');
00171 
00172       XMLCustomWriter::createChildWithText($doc, $root, 'name', $author->getFullName());
00173       XMLCustomWriter::createChildWithText($doc, $root, 'email', $author->getEmail(), false);
00174 
00175       if(in_array($author->getLocalizedAffiliation(), $affilList)  && !empty($affilList[0])) {
00176          XMLCustomWriter::createChildWithText($doc, $root, 'affiliationId', current(array_keys($affilList, $author->getLocalizedAffiliation())));
00177       }
00178       
00179       return $root;
00180    }
