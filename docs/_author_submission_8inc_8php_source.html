00150    /**
00151     * Get the submission status. Returns one of the defined constants
00152     * (STATUS_INCOMPLETE, STATUS_ARCHIVED, STATUS_PUBLISHED,
00153     * STATUS_DECLINED, STATUS_QUEUED_UNASSIGNED, STATUS_QUEUED_REVIEW,

00157     * getSubmissionStatus in the SectionEditorSubmission class and
00158     * changes here should be propagated.
00159     */
00160    function getSubmissionStatus() {
00161       $status = $this->getStatus();
00162       if ($status == STATUS_ARCHIVED || $status == STATUS_PUBLISHED ||
00163           $status == STATUS_DECLINED) return $status;
00164 
00165       // The submission is STATUS_QUEUED or the author's submission was STATUS_INCOMPLETE.
00166       if ($this->getSubmissionProgress()) return (STATUS_INCOMPLETE);
00167 
00168       // The submission is STATUS_QUEUED. Find out where it's queued.
00169       $editAssignments = $this->getEditAssignments();
00170       if (empty($editAssignments))
00171          return (STATUS_QUEUED_UNASSIGNED);
00172 
00173       $latestDecision = $this->getMostRecentDecision();
00174       if ($latestDecision) {
00175          if ($latestDecision == SUBMISSION_EDITOR_DECISION_ACCEPT || $latestDecision == SUBMISSION_EDITOR_DECISION_DECLINE) {
00176             return STATUS_QUEUED_EDITING;
00177          }
00178       }
00179       return STATUS_QUEUED_REVIEW;
00180    }
