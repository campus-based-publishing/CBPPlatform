00168    /**
00169     * Claims (locks) the next raw (unparsed) citation found in the
00170     * database and checks it. This method is idempotent and parallelisable.
00171     * It uses an atomic locking strategy to avoid race conditions.
00172     *
00173     * @param $request Request
00174     * @param $lockId string a globally unique id that
00175     *  identifies the calling process.
00176     * @return boolean true if a citation was found and checked, otherwise
00177     *  false.
00178     */
00179    function checkNextRawCitation(&$request, $lockId) {
00180       // NB: We implement an atomic locking strategy to make
00181       // sure that no two parallel background processes can claim the
00182       // same citation.
00183       $rawCitation = null;
00184       for ($try = 0; $try < 3; $try++) {
00185          // We use three statements (read, write, read) rather than
00186          // MySQL's UPDATE ... LIMIT ... to guarantee compatibility
00187          // with ANSI SQL.
00188 
00189          // Get the ID of the next raw citation.
00190          $result =& $this->retrieve(
00191             'SELECT citation_id
00192             FROM citations
00193             WHERE citation_state = ?
00194             LIMIT 1',
00195             CITATION_RAW
00196          );
00197          if ($result->RecordCount() > 0) {
00198             $nextRawCitation = $result->GetRowAssoc(false);
00199             $nextRawCitationId = $nextRawCitation['citation_id'];
00200          } else {
00201             // Nothing to do.
00202             $result->Close();
00203             return false;
00204          }
00205          $result->Close();
00206          unset($result);
00207 
00208          // Lock the citation.
00209          $this->update(
00210             'UPDATE citations
00211             SET citation_state = ?, lock_id = ?
00212             WHERE citation_id = ? AND citation_state = ?',
00213             array(CITATION_CHECKED, $lockId, $nextRawCitationId, CITATION_RAW)
00214          );
00215 
00216          // Make sure that no other concurring process
00217          // has claimed this citation before we could
00218          // lock it.
00219          $result =& $this->retrieve(
00220             'SELECT *
00221             FROM citations
00222             WHERE lock_id = ?',
00223             $lockId
00224          );
00225          if ($result->RecordCount() > 0) {
00226             $rawCitation =& $this->_fromRow($result->GetRowAssoc(false));
00227             break;
00228          }
00229       }
00230       $result->Close();
00231       if (!is_a($rawCitation, 'Citation')) return false;
00232 
00233       // Check the citation.
00234       $filteredCitation =& $this->checkCitation($request, $rawCitation);
00235 
00236       // Updating the citation will also release the lock.
00237       $this->updateObject($filteredCitation);
00238 
00239       return true;
00240    }

00308       //    should not be included then return only non-optional
